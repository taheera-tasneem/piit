    "## Functions in Pandas\n",
    "- Understanding Data\n",
    "    - type  # type of object\n",
    "    - shape # rows & cols in data\n",
    "    - columns # column names\n",
    "    - dtypes  # datatype \n",
    "    - info()  # datatypes & non-null counts\n",
    "    - sample(x) # randon x rows\n",
    "    - head(x) # first x rows\n",
    "    - tail(x) # last x rows\n",
    "- Summary Statistics\n",
    "    - describe()  #summary stats for numeric cols\n",
    "    - describe(include='all')  #summary stats for all cols\n",
    "    - mtcars.nunique() #count of unique values per column\n",
    "    - value_counts()  # freq count for a series\n",
    "- Missing/Duplicates\n",
    "    -  isnull().sum()   # missing value count per cols\n",
    "    -  notnull().sum()  # non missing value count\n",
    "    -  duplicated(().sum() # count of duplicate rows\n",
    "- View Specific Col/ Row\n",
    "    -  ['col_name']   # single column\n",
    "    -  [['col1','col2']] # multiple columns\n",
    "    -  iloc[0]   #1st row by index position\n",
    "    -  loc[0] #1st row by label\n",
    "    -  iloc[0:5]   # first 5 rows\n",
    "-  Basic Data Insights\n",
    "    -  corr(numeric_only=True)  #correlation matrix (numeric only)\n",
    "    -  mean(numeric_only=True)   #mean of numeric cols\n",
    "    -  mode().iloc(0) #most frequent values\n",
    "    -  median(numeric_only=True) # median of numeric cols "
